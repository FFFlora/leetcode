class Solution:
        
    def findKthLargest(self, nums: List[int], k: int) -> int:
        start = 0
        end = len(nums) - 1
        return self.quickSelect(nums,start,end,k)

    def quickSelect(self,A,start,end,k):

        Pindex = self.partition(A,start,end)
        if Pindex == k-1:
            ans = len(A) - k    
            return A[ans]
        elif Pindex >= k:
            return self.quickSelect(A,start,Pindex-1,k)
        return self.quickSelect(A,Pindex+1,end,k)


    def partition(self, A,start,end):
        pivot = A[end]
        Pindex  = start 
        for i in range(start,end):
            if (A[i] <= pivot):
                A[i],A[Pindex] = A[Pindex],A[i]
                Pindex = Pindex + 1
        A[Pindex], A[end] = A[end], A[Pindex]
        return Pindex
